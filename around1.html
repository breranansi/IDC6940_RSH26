# Build Reverse Box-Cox Transform
def unBox(zee,lam):
    # zee = transformed variable
    # lam = box-cox parameter
    if (lam  != 0.0):
        return np.pow((lam*zee+1.0),(1.0/lam))
    else:
        return np.exp(zee)

# Build Forward Box-Cox Transform
def reBox(why,lam):
    # why = untransformed variable
    # lam = box-cox parameter
    if (lam  != 0.0):
        return (np.pow(why,lam)-1.0)/lam
    else:
        return np.log(why)
